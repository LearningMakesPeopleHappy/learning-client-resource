# 67. 二进制求和

## 题目

给你两个二进制字符串，返回它们的和（用二进制表示）。

输入为 **非空** 字符串且只包含数字 `1` 和 `0`。

```javascript
/**
 * @param {string} a
 * @param {string} b
 * @return {string}
 */
var addBinary = function(a, b) {
    let max, min;
    if (a.length >= b.length) {
        max = a;
        min = b;
    } else {
        max = b;
        min = a;
    }

    const maxLen = max.length;
    const minLen = min.length;
    let sum = '';
    let need = false;
    let i = 1;
    for (; i <= minLen; i++) {
        const val = max[maxLen - i] + min[minLen - i] + (need ? '1' : '0')
        switch (val) {
            case '111':
                sum = '1' + sum;
                need = true;
                break;
            case '011':
            case '101':
            case '110':
                sum = '0' + sum;
                need = true;
                break;
            case '001':
            case '010':
            case '100':
                sum = '1' + sum;
                need = false;
                break;
            case '000':
                sum = '0' + sum;
                need = false;
                break;
        }
    }

    if (need) {
        while (max[maxLen - i] === '1') {
            sum = '0' + sum;
            i++;
        }

        sum = '1' + sum;
        i++;
    }
    
    return i > maxLen ? sum : max.slice(0, maxLen - i + 1) + sum;
};
```
