# 1. 合并两个有序链表

## 题目

将两个升序链表合并为一个新的升序链表并返回。新链表是通过拼接给定的两个链表的所有节点组成的。 

> **测试**
>
> [1,2,4]
> [1,3,4]
>
> [1, 1, 2, 3, 4, 4]

#### 代码

```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} l1
 * @param {ListNode} l2
 * @return {ListNode}
 */
var mergeTwoLists = function(l1, l2) {
    if (!l1 && !l2) return null;
    else if (!l1) return l2; 
    else if (!l2) return l1;

    let start;
    let current;
    let node1 = l1;
    let node2 = l2;
    while (node1 && node2) {
        if (node1.val <= node2.val) {
            if (start) {
                current.next = node1
                current = node1;
                node1 = node1.next;
            } else {
                start = node1;
                current = node1;
                node1 = node1.next;
            }
        } else {
            if (start) {
                current.next = node2;
                current = node2;
                node2 = node2.next;
            } else {
                start = node2;
                current = node2;
                node2 = node2.next;
            }
        }
    }

    if (node1) {
        current.next = node1;
    } else if (node2) {
        current.next = node2;
    }

    current = null;

    return start;
};
```

#### 代码

```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} l1
 * @param {ListNode} l2
 * @return {ListNode}
 */
var mergeTwoLists = function(l1, l2) {
    if (!l1 && !l2) return null;
    else if (!l1) return l2; 
    else if (!l2) return l1;

    if (l1.val <= l2.val) {
        l1.next = mergeTwoLists(l1.next, l2);
        return l1;
    } else {
        l2.next = mergeTwoLists(l1, l2.next);
        return l2;
    }
};
```
